101void Shell_Loop(void){
102 if(uartRxReceived){
103 switch(uartRxBuffer[0]){
104 case ASCII_CR: // Nouvelle ligne, instruction à traiter
105 HAL_UART_Transmit(&huart2, newline, sizeof(newline), HAL_MAX_DELAY);
106 cmdBuffer[idx_cmd] = '\0';
107 argc = 0;
108 token = strtok(cmdBuffer, " ");
109 while(token!=NULL){
110 argv[argc++] = token;
111 token = strtok(NULL, " ");
112 }
113 idx_cmd = 0;
114 newCmdReady = 1;
115 break;
116 case ASCII_BACK: // Suppression du dernier caractère
117 cmdBuffer[idx_cmd--] = '\0';
118 HAL_UART_Transmit(&huart2, backspace, sizeof(backspace), HAL_MAX_DELAY);
119 break;
120
121 default: // Nouveau caractère
122 cmdBuffer[idx_cmd++] = uartRxBuffer[0];
123 HAL_UART_Transmit(&huart2, uartRxBuffer, UART_RX_BUFFER_SIZE, HAL_MAX_DELAY);
124 }
125 uartRxReceived = 0;
126 }

128 if(newCmdReady){
